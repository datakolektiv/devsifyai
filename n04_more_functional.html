Lapply(), Sapply(), Apply()

### devsify.ai
### Notes for Session04 More Functional Programming

v1 <- 1:100

# - power() function
power <- function(x) {
  # code blocks...
  return(x^2)
}

# - silly functional:
squares <- lapply(v1, power)
class(squares)
## [1] "list"
# - sapply():
squares <- sapply(v1, power)
class(squares)
## [1] "numeric"
# - apply
m1 <- matrix(data = c(1,2,3,4),
             ncol = 2)
sum(c(1,2))
## [1] 3
apply(X = m1, MARGIN = 1, FUN = sum)
## [1] 4 6
apply(X = m1, MARGIN = 2, FUN = sum)
## [1] 3 7
